Sample Code


$(document).ready(function(){
	//함수 호출
	SetCarousel();
	SetTimer();
});
//함수는 ready()함수 밖에서 정의
//자바스크립트함수 정의
function SetCarousel() {
	 $("#carousel #imageContainer").first().addClass("selected");
}

function SetTimer() {
	setInterval(showNext, 2000);
	//setInterval(showNext, 2000);함수는 자바스크립트 내장 함수 showNext는 실행할 함수, 2000은 정지 시간
	//2000밀리세컨드 후에 showNext()를 실행
}

function showNext() {
	var sel = $("#carousel #imageContainer.selected"); //엘리먼트 객체를 선택 해서 변수 sel에 저장
	sel.removeClass("selected"); //sel객체에서 class속성 제거
	var nextImge = sel.next(); //다음 sel객체 선택해서 변수 nexrInage에 저장
	var id = $(nextImge).attr("id"); //속성 id가 있는 nextImge객체 선택해서 변수 id에 저장
    if (id != null) {
        nextImge.addClass("selected"); //객체id가 있는 경우
    }
    else {
    	SetCarousel();
    }
}
